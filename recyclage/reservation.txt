# Inscription, Connexion & Deconnexion.
def inscription(request):
    if request.method == 'POST':
        form = InscriptionForm(request.POST)

        if form.is_valid():
            user = form.save()

            Client.objects.create(
                user=user,
                telephone=""
            )

            return redirect('connexion')
    else:
        form = InscriptionForm()

    return render(request, 'inscription.html', {'form': form})


def connexion(request):
    if request.method == 'POST':
        username = request.POST.get('username')
        password = request.POST.get('password')

        user = authenticate(
            request,
            username=username,
            password=password
        )

        if user is not None:
            login(request, user)
            return redirect('dashboard')
        else:
            return render(
                request,
                'connexion.html',
                {'erreur': 'Nom d’utilisateur ou mot de passe incorrect'}
            )

    return render(request, 'connexion.html')




def deconnexion(request):
    logout(request)
    return redirect('connexion')

@login_required
def reservation_vehicule(request, vehicule_id):
    vehicule = get_object_or_404(Vehicule, id=vehicule_id)
    
    if request.method == 'POST':
        form = ReservationForm(request.POST)
        if form.is_valid():
            reservation = form.save(commit=False)
            reservation.user = request.user
            reservation.vehicule = vehicule
            # --- CALCUL DU PRIX TOTAL CÔTÉ SERVEUR ---
            # 1. Récupérer le nombre de jours
            delta_jours = (reservation.date_fin - reservation.date_debut).days
            if delta_jours <= 0:
                delta_jours = 1 # Sécurité au cas où c'est la même journée
                
            # 2. Calcul de base (Prix du véhicule x Jours)
            total = vehicule.prix_par_jour * delta_jours
            
            # 3. Ajouter l'option chauffeur si cochée
            avec_chauffeur = "chauffeur" in request.POST
            reservation.inclut_chauffeur = avec_chauffeur
            if avec_chauffeur:
                total += 10000  # Option forfaitaire ou (10000 * delta_jours) selon votre choix
                
            # Enregistrer le total calculé dans le modèle
            reservation.prix_total = total
            # -----------------------------------------
            
            overlapping = Reservation.objects.filter(
                vehicule=vehicule,
                date_debut__lt=reservation.date_fin,
                date_fin__gt=reservation.date_debut
            )

            if overlapping.exists():
                form.add_error(None, "Ce véhicule est déjà réservé pour ces dates.")
            else:
                reservation.save()
                return redirect('page_paiement', reservation_id=reservation.id)
        else:
            # Si le formulaire n'est pas valide, on peut voir les erreurs dans le terminal
            print(form.errors)
            
    else:
        form = ReservationForm()

    return render(request, 'reservation.html', {
        'form': form,
        'vehicule': vehicule
    })

@login_required
def modifier_profil(request):

    if request.method == 'POST':
        form = ProfilForm(
            request.POST,
            instance=request.user
        )

        if form.is_valid():
            form.save()

            request.user.client.telephone = form.cleaned_data['telephone']
            request.user.client.save()

            return redirect('dashboard')

    else:
        form = ProfilForm(instance=request.user)

        form.fields['telephone'].initial = (
            request.user.client.telephone
        )

    return render(
        request,
        'modifier_profil.html',
        {'form': form}
    )

# Adaptateur pour forcer TLSv1.2 (requis par l'infrastructure bancaire)
class TLSAdapter(HTTPAdapter):
    def init_poolmanager(self, *args, **kwargs):
        context = ssl.create_default_context()
        context.minimum_version = ssl.TLSVersion.TLSv1_2
        kwargs['ssl_context'] = context
        return super(TLSAdapter, self).init_poolmanager(*args, **kwargs)

@login_required
def initier_paiement(request, reservation_id):
    reservation = get_object_or_404(Reservation, id=reservation_id, user=request.user)
    
    if request.method == 'POST':
        # Format d'identifiant unique proche de l'exemple BCEAO
        tx_id = f"PAY-{reservation.id}-{uuid.uuid4().hex[:4].upper()}"
        
        # --- ÉTAPE 1 : OBTENTION DU TOKEN OAUTH2 ---
        auth_data = {
            "grant_type": "client_credentials",
            "scope": "piz/paiement.write"
        }
        auth_headers = {
            "Content-Type": "application/x-www-form-urlencoded"
        }
        
        try:
            auth_response = requests.post(
                settings.BCEAO_AUTH_URL, 
                data=auth_data, 
                headers=auth_headers,
                auth=HTTPBasicAuth(settings.BCEAO_CLIENT_ID, settings.BCEAO_CLIENT_SECRET),
                timeout=10
            )
            
            if auth_response.status_code != 200:
                return render(request, 'paiement.html', {
                    'reservation': reservation, 
                    'error': "Échec de la validation des identifiants auprès d'Amazon Cognito (BCEAO)."
                })
                
            token = auth_response.json().get("access_token")
            
            # --- ÉTAPE 2 : ENVOI DE LA DEMANDE DE PAIEMENT (FORMAT STRICT CURL) ---
            headers = {
                "Authorization": f"Bearer {token}",
                "Content-Type": "application/json",
                "Accept": "application/json"
            }
            
            # Copie conforme du dictionnaire d'après votre exemple curl
            payload = {
                "txId": tx_id,
                "confirmation": False,
                "beneficiaireAlias": "9b1b3499-3e50-435b-b757-ac7a83d8aa96", # ID Marchand de test
                "montant": int(reservation.prix_total),
                "motif": f"Paiement reservation KAMSA Auto #{reservation.id}"
            }
            
            print(f"--- [ÉTAPE 2] ENVOI DU PAIEMENT (txId: {tx_id}) ---")
            
            # Application de la session avec l'adaptateur TLS sécurisé
            session = requests.Session()
            session.mount("https://", TLSAdapter())
            
            payment_response = session.post(
                settings.BCEAO_PAYMENT_URL,  # Assurez-vous qu'elle vaut "https://sandbox.api.pi-bceao.com/piz/v1/paiements"
                json=payload, 
                headers=headers, 
                timeout=15
            )
            
            print(f"Statut Paiement BCEAO : {payment_response.status_code}")
            print("Réponse brute Paiement :", payment_response.text)
            
            if payment_response.status_code in [200, 201]:
                # Sauvegarde des données en cas de succès du simulateur
                reservation.transaction_id = tx_id
                reservation.save()
                return redirect('paiement_succes')
            else:
                return render(request, 'paiement.html', {
                    'reservation': reservation, 
                    'error': f"Erreur de traitement BCEAO (Code HTTP {payment_response.status_code})."
                })
                
        except requests.exceptions.RequestException as e:
            print(f"❌ Erreur réseau système : {e}")
            return render(request, 'paiement.html', {
                'reservation': reservation, 
                'error': f"Impossible de joindre le serveur de la banque centrale : {str(e)}"
            })
            
    return redirect('page_paiement', reservation_id=reservation.id)
